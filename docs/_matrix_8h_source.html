377 // 현재 행렬의 역행렬(Inverse Matrix)을 반환함
378 // 역행렬이 존재하지 않으면 Identity를 반환함
379 // 현재는 역행렬이 없을 때 Identity를 반환/대입하는 정책입니다.
380 // 디버깅 투명성을 높이려면 원본 유지 + false 반환 정책도 고려할 수 있습니다.
381 FMatrix GetInverse(float Tolerance = 1.e-8f) const noexcept
382 {
383 const DirectX::XMMATRIX XM = ToXMMatrix();
384
385 DirectX::XMVECTOR Det;
386 const DirectX::XMMATRIX Inv = DirectX::XMMatrixInverse(&Det, XM);
387
388 const float DeterminantValue = DirectX::XMVectorGetX(Det);
389 if (std::fabs(DeterminantValue) <= Tolerance)
390 {
391 // 역행렬이 없을 때 원본으로 반환하는 정책으로 변경 가능
392#ifndef NDEBUG
393 assert("FMatrix::GetInverse() failed: matrix is singular or invalid.");
394#endif
395 return Identity;
396 }
397
398 return FMatrix(Inv);
399 }

405 [[nodiscard]] bool Inverse(float Tolerance = 1.e-8f) noexcept

408

479 // 현재 행렬을 위치, 회전 행렬, 스케일로 분해함
480 // 분해에 실패하면 false를 반환함
481 bool Decompose(FVector& OutTranslation, FMatrix& OutRotation, FVector& OutScale, float Tolerance = 1.e-8f) const noexcept
482 {
483 // 1차 버전의 함수
484 // 다음 경우까지 완벽하게 다루지 않음
485 // - negative scale
486 // - reflection
487 // - shear
488 // - 축이 서로 완전히 직교하지 않은 행렬
489 // 수학적으로 보든 4x4 행렬을 완벽하게 분해하는 구현은 아님.
490 OutTranslation = GetOrigin();
491
492 const FVector XAxis = GetScaledAxis(EAxis::X);
493 const FVector YAxis = GetScaledAxis(EAxis::Y);
494 const FVector ZAxis = GetScaledAxis(EAxis::Z);
495
496 OutScale = FVector(XAxis.Size(), YAxis.Size(), ZAxis.Size());
497
498 if (OutScale.X <= Tolerance || OutScale.Y <= Tolerance || OutScale.Z <= Tolerance)
499 {
500 OutRotation = Identity;
501 return false;
502 }
503
504 const FVector UnitX = XAxis / OutScale.X;
505 const FVector UnitY = YAxis / OutScale.Y;
506 const FVector UnitZ = ZAxis / OutScale.Z;
507
508 OutRotation = Identity;
509 OutRotation.SetAxes(UnitX, UnitY, UnitZ, FVector::ZeroVector);
510
511 return true;
512 }

602 {

605 {
606 return Identity;
607 }
608

696 // Left-Handed 기준 원근 투영 행렬을 생성함
697 static FMatrix MakePerspectiveFovLH(float FovYRad, float AspectRatio, float NearZ, float FarZ) noexcept
698 {
699 assert(AspectRatio != 0.f);
700 assert(FarZ != NearZ);
701
702 const float YScale = 1.0f / std::tan(FovYRad * 0.5f);
703 const float XScale = YScale / AspectRatio;
704
705 return FMatrix(
706 0.f, 0.f, FarZ / (FarZ - NearZ), 1.f,
707 XScale, 0.f, 0.f, 0.f,
708 0.f, YScale, 0.f, 0.f,
709 0.f, 0.f, -NearZ * FarZ / (FarZ - NearZ), 0.f
710 );

781 // 위치, 회전 행렬, 스케일을 이용하여 월드 행렬을 생성함
782 // TODO : FRotator, FQuat를 만들면 수정해야함.
783 static FMatrix MakeWorld(const FVector& Translation, const FMatrix& RotationMatrix, const FVector& Scale) noexcept
784 {
785 FMatrix Result = RotationMatrix;
786
787 Result.M[0][0] *= Scale.X;
788 Result.M[0][1] *= Scale.X;
789 Result.M[0][2] *= Scale.X;
790
791 Result.M[1][0] *= Scale.Y;
792 Result.M[1][1] *= Scale.Y;
793 Result.M[1][2] *= Scale.Y;
794
795 Result.M[2][0] *= Scale.Z;
796 Result.M[2][1] *= Scale.Z;
797 Result.M[2][2] *= Scale.Z;
798
799 Result.M[3][0] = Translation.X;
800 Result.M[3][1] = Translation.Y;
801 Result.M[3][2] = Translation.Z;
802 Result.M[3][3] = 1.f;
803
804 return Result;
805 }
